##### Trade Initalizaion #######
country_event = {
	id = tradelite.00
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = {
				has_global_flag = trader_lite_active
			}
			every_planet = {
				remove_planet_flag = has_tier_1
				remove_planet_flag = has_tier_2
				remove_planet_flag = has_tier_3
				remove_planet_flag = has_tier_4
				remove_planet_flag = has_tier_5
			}
		}
		set_global_flag = trader_lite_active
		event_target:trader_country = {
			every_owned_ship = {
				fleet = {
					destroy_fleet = this
				}
			}
		}
		every_country = {
			limit = {
				any_owned_planet = {
					AND = {
						has_spaceport = yes
						OR = {
							has_spaceport_module = trade_station_1
							has_spaceport_module = trade_station_2
							has_spaceport_module = trade_station_3
							has_spaceport_module = trade_station_4
							has_spaceport_module = trade_station_5
						}
					}
				}
			}
			random = {
				chance = 85
				country_event = { id = tradelite.1 days = 1 }
			}
		}
	}
}

country_event = {
	id = tradelite.1
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				not = {
					has_country_flag = traderlite_running
				}
				has_global_flag = trader_lite_active
			}
			set_country_flag  = traderlite_running
			random_owned_planet = {
				limit = { 
					OR = {
						has_spaceport_module = "trade_station_1"
						has_spaceport_module = "trade_station_2"
						has_spaceport_module = "trade_station_3"
						has_spaceport_module = "trade_station_4"
						has_spaceport_module = "trade_station_5"
					}
					NOT = { has_planet_flag = trader_depleted_flag }
				}
				planet_event = {
					id = tradelite.10
					days = 1
				}
			}
		}
	}
	option = {
		name = ok
	}
}
